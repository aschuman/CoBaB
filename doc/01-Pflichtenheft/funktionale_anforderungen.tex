\subsection{Pflichtanforderungen}
\begin{description}
	\item[\req{F 10}] Starten des Programms
	\item[\req{F 20}] Beenden des Programms
	\item[\req{F 30}] Automatisches Erkennen verfügbarer Suchverfahren
	\item[\req{F 40}] Anzeigen eines Hilfe-Dialogs mit Hinweisen zur Benutzung des Programms
	\item[\req{F 50}] Anzeigen eines About-Dialogs mit Informationen zum Programm
	\newline
%	\item[\req{F 0}] Er\"offnen eines neuen Suchkontexts
%	\item[\req{F 0}] Er\"offnen eines neuen \glslink{Suchkontext}{Suchkontexts}
 
	\item[\req{F 80}] Anzeigen einer Bibliothek von Datensätzen, wie in Abbildung \ref{fig:bibliothek} zu sehen ist
	\item[\req{F 90}] Hinzufügen von Datensätzen zur Bibliothek, wie in Abbildung \ref{fig:filechooser} zu sehen ist
	\item[\req{F 100}] Ausw\"ahlen eines Datensatzes aus der Bibliothek
	\item[\req{F 110}] Anzeigen einer Übersicht der Bilder bzw. Videos des gewählten Datensatzes, wie in Abbildung \ref{fig:fotoanzeiger} zu sehen ist
	\item[\req{F 115}] Unterstützung der Bildformate JPEG, PNG und BMP
	\item[\req{F 120}] Anzeigen einer größeren Darstellung eines ausgewählten Bildes bzw. Videos des Datensatzes
	\item[\req{F 130}] Abspielen eines Videos aus Einzelbildern
	\item[\req{F 135}] Auswählen des vorherigen oder nächsten Bildes bzw. Videos für die große Ansicht % Buttons ist keine Funktion
	\item[\req{F 140}] Ausw\"ahlen eines Bildes bzw. Videos aus dem gewählten Datensatz als Suchvorlage
	\item[\req{F 150}] Anzeigen \glslink{Annotation}{annotierter Bildbereiche}
	\item[\req{F 160}] Auswahl eines \glslink{Annotation}{annotierten Bildbereiches} als Suchvorlage
	\item[\req{F 170}] Einschränkung der Suchvorlage auf einen benutzerdefinierten rechteckigen Bereich (falls keine \gls{Annotation} gewählt wurde)
	\item[\req{F 180}] Auswahl eines für ggf. gewählte \gls{Annotation} geeigneten Suchverfahrens
	\item[\req{F 190}] Anzeigen einer Beschreibung für die Suchverfahren
	\item[\req{F 200}] Festlegen der für das gewählte Suchverfahren spezifischen Parameter, wie in Abbildung \ref{fig:parameterauswahl} zu sehen ist
	\item[\req{F 210}] Anzeigen sämtlicher vorgenommener Einstellungen zur aktuellen Suche, um Überprüfung durch den Benutzer zu ermöglichen, wie in Abbildung \ref{fig:überprüfung} zu sehen ist
	\item[\req{F 220}] Ändern sämtlicher vorgenommener Einstellungen nach der Überprüfung
	\newline
	\item[\req{F 230}] Starten der Suche
	\item[\req{F 240}] Anzeigen einer Fortschrittsanimation während des Suchvorgangs

	\newline
	\item[\req{F 260}] Anzeigen einer \"Ubersicht der Suchergebnisse, wie in Abbildung \ref{fig:suchergebnisse}
	\item[\req{F 290}] Einstellen eines Feedbacks zu einem Suchergebnis und Starten einer weiteren Suche mit diesem Feedback und dem selben Suchverfahren
	
	\item[\req{F 320}] Loggen von Informationen, Warnungen, Fehlern und Debugmeldungen
	 
\end{description}

\subsection{Wunschanforderungen}
\begin{description}
	\item[\req{FW 10}] nicht-flüchtige Auswahl der Übersetzungen der \gls{GUI} in Deutsch und Englisch
	\newline
	\item[\req{FW 20}] Wählen einer Suchvorlage, die nicht in den gewählten Datensätzen enthalten ist (erweitert \req{F 140})
	\item[\req{FW 25}] Wählen von mehreren Datensätzen in denen gesucht wird (erweitert \req{F 100})
	\newline
	\item[\req{FW 30}] Abspielen eines Benachrichtigungstons bei Abschluss einer Suche
	\item[\req{FW 40}] Ein-/Abschalten des Benachrichtigungstons
	\item[\req{FW 50}] Abspielen von Videos aus Videodateien (mit Ausgabe der Audiospur)
	\item[\req{FW 60}] Durchführen einer weiteren Suche in den Ergebnissen einer bereits beendeten Suche mit anderem Suchverfahren \req{F 290}
	\newline
	\item[\req{FW 70}] Wechsel in einen Vollbildmodus für Präsentation
	
	\item[\req{FW 80}] Zoomen und Scrollen in der größeren Darstellung in der Datensatzauswahl
	\item[\req{FW 90}] Generierung eines zufälligen oder festgelegten Vorschaubildes für den Datensatz
	\item[\req{FW 100}] Anzeigen einer größeren Darstellung eines ausgewählten Suchergebnisses
	
	\item[\req{F 310}] nicht-flüchtiges Speichern der Lesezeichen (siehe auch \req{F 60})
	\item[\req{F 315}] automatisches nicht-flüchtiges Speichern der Chronik (siehe auch \req{F 65})
	\item[\req{FW 70}] Anzeigen einer Übersicht der Lesezeichen
	\item[\req{FW 75}] Anzeigen einer Übersicht der Chronik
	\item[\req{FW 60}] Wiederanzeigen der Ergebnisse eines Lesezeichens (siehe auch \req{F 310})
	\item[\req{FW 65}] Wiederanzeigen der Ergebnisse der Chronik (siehe auch \req{F 315})
	
	\item[\req{F 250}] Abbrechen der Suche mit Weiterführung der Oberfläche
	\item[\req{FW 110}] Festlegung eines Wertes zwischen 0 und 10 als Feedback
\end{description}
